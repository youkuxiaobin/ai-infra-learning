# 04 分块和不规则访存

## 1. GEMM 的乘法代码

### naive 版本

```cpp
template<typename T>
__global__ void gemm_naive(int M, int N, int K, float alpha,
                           const T *A, const T *B, float beta,
                           T *C) {
    int row = blockIdx.y * blockDim.y + threadIdx.x; // Row of C
    int col = blockIdx.x * blockDim.x + threadIdx.y; // Column of C

    if (row < M && col < N) {
        float sum = 0.0f;
        // Inner loop over K (dot product for C[row][col])
        for (int k = 0; k < K; ++k) {
            sum += A[row * K + k] * B[k * N + col];
        }
        C[row * N + col] = alpha * sum + beta * C[row * N + col];
    }
}
```

假设这个 warp 负责：

- `col = 7`
- `row = 0..31`

## 2. naive 版本的访存分析

### 看 A 的访问

`Naive: 访问 A[row * K + k]`，固定 `k = 5`

| 线程 | row | col | 访问的 A 元素 | 一维地址 |
|---|---:|---:|---|---|
| t0 | 0 | 7 | A[0][5] | A[5] |
| t1 | 1 | 7 | A[1][5] | A[1029] |
| t2 | 2 | 7 | A[2][5] | A[2053] |
| t3 | 3 | 7 | A[3][5] | A[3077] |
| ... | ... | ... | ... | ... |
| t31 | 31 | 7 | A[31][5] | A[31749] |

结论：

- 不是连续
- 不是同地址
- 而是**超大跨步访问**
- 这是很差的访存模式

### 看 B 的访问

`Naive: 访问 B[k * N + col]`，固定 `k = 5`

| 线程 | row | col | 访问的 B 元素 | 一维地址 |
|---|---:|---:|---|---|
| t0 | 0 | 7 | B[5][7] | B[5127] |
| t1 | 1 | 7 | B[5][7] | B[5127] |
| t2 | 2 | 7 | B[5][7] | B[5127] |
| ... | ... | ... | ... | ... |
| t31 | 31 | 7 | B[5][7] | B[5127] |

结论：

- 这里是**同地址重复读取**
- 这一项不算太坏

### 看 C 的写回

`Naive: 写回 C[row * N + col]`

| 线程 | row | col | 写入的 C 元素 | 一维地址 |
|---|---:|---:|---|---|
| t0 | 0 | 7 | C[0][7] | C[7] |
| t1 | 1 | 7 | C[1][7] | C[1031] |
| t2 | 2 | 7 | C[2][7] | C[2055] |
| ... | ... | ... | ... | ... |
| t31 | 31 | 7 | C[31][7] | C[31751] |

结论：

- 写回也是**大跨步**
- 很不友好

## 3. coalesced 版本

```cpp
template<typename T>
__global__ void gemm_coalesced(int M, int N, int K,
                               T alpha,
                               const T* A,
                               const T* B,
                               T beta,
                               T* C) {
    // blockIdx: 当前 block 在整个 grid 中的位置
    // blockDim: 每个 block 的尺寸
    // threadIdx: 当前线程在 block 内的位置

    // 计算这个线程对应 C 中的行号
    int row = blockIdx.y * blockDim.y + threadIdx.y;

    // 计算这个线程对应 C 中的列号
    int col = blockIdx.x * blockDim.x + threadIdx.x;

    // 只有落在合法矩阵范围内的线程才参与计算
    if (row < M && col < N) {
        // 用来累加点积结果
        T sum = 0;

        // C[row][col] = sum(A[row][k] * B[k][col]), k=0..K-1
        for (int k = 0; k < K; ++k) {
            T a = A[row * K + k]; // A[row][k]
            T b = B[k * N + col]; // B[k][col]
            sum += a * b;
        }

        // 最终写回
        C[row * N + col] = alpha * sum + beta * C[row * N + col];
    }
}
```

## 4. coalesced 版本的访存分析

### 看 A 的访问

内层循环某一次固定 `k = 5`：

- `row = 7` 固定
- `k = 5` 固定
- `A[7 * 1024 + 5] = A[7173]`

`Coalesced: 访问 A[row * K + k]`，固定 `k = 5`

| 线程 | row | col | 访问的 A 元素 | 一维地址 |
|---|---:|---:|---|---|
| t0 | 7 | 0 | A[7][5] | A[7173] |
| t1 | 7 | 1 | A[7][5] | A[7173] |
| t2 | 7 | 2 | A[7][5] | A[7173] |
| t3 | 7 | 3 | A[7][5] | A[7173] |
| ... | ... | ... | ... | ... |
| t31 | 7 | 31 | A[7][5] | A[7173] |

结论：

- 不是连续地址
- 而是 **32 个线程都读同一个地址**

### 看 B 的访问

`B[k * N + col]`

- `B[5 * 1024 + col]`
- warp 内 `col = 0..31`

`Coalesced: 访问 B[k * N + col]`，固定 `k = 5`

| 线程 | row | col | 访问的 B 元素 | 一维地址 |
|---|---:|---:|---|---|
| t0 | 7 | 0 | B[5][0] | B[5120] |
| t1 | 7 | 1 | B[5][1] | B[5121] |
| t2 | 7 | 2 | B[5][2] | B[5122] |
| t3 | 7 | 3 | B[5][3] | B[5123] |
| ... | ... | ... | ... | ... |
| t31 | 7 | 31 | B[5][31] | B[5151] |

结论：

- 32 个线程访问连续地址
- 这就是典型的 **coalesced load**

### 看 C 的写回

`C[row * N + col]`

- `row = 7`
- `col = 0..31`

`Coalesced: 写回 C[row * N + col]`

| 线程 | row | col | 写入的 C 元素 | 一维地址 |
|---|---:|---:|---|---|
| t0 | 7 | 0 | C[7][0] | C[7168] |
| t1 | 7 | 1 | C[7][1] | C[7169] |
| t2 | 7 | 2 | C[7][2] | C[7170] |
| ... | ... | ... | ... | ... |
| t31 | 7 | 31 | C[7][31] | C[7199] |

## 5. 先理解 GEMM 到底在算什么

完整公式：

```text
C = αAB + βC
```

### launch 写法

```cpp
dim3 blockDim(16, 16);
dim3 gridDim((N + blockDim.x - 1) / blockDim.x,
             (M + blockDim.y - 1) / blockDim.y);

gemm_coalesced<float><<<gridDim, blockDim>>>(M, N, K,
                                              1.0f,
                                              A, B,
                                              0.0f,
                                              C);
```

## 6. 完整代码

```cpp
#include <cuda_runtime.h>
#include <iostream>
#include <vector>

template<typename T>
__global__ void gemm_coalesced(int M, int N, int K,
                               T alpha,
                               const T* A,
                               const T* B,
                               T beta,
                               T* C) {
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;

    if (row < M && col < N) {
        T sum = 0;
        for (int k = 0; k < K; ++k) {
            sum += A[row * K + k] * B[k * N + col];
        }
        C[row * N + col] = alpha * sum + beta * C[row * N + col];
    }
}

int main() {
    int M = 2;
    int K = 3;
    int N = 2;

    std::vector<float> h_A = {
        1, 2, 3,
        4, 5, 6
    }; // 2x3

    std::vector<float> h_B = {
        7, 8,
        9, 10,
        11, 12
    }; // 3x2

    std::vector<float> h_C(M * N, 0.0f); // 2x2

    float *d_A, *d_B, *d_C;
    cudaMalloc(&d_A, M * K * sizeof(float));
    cudaMalloc(&d_B, K * N * sizeof(float));
    cudaMalloc(&d_C, M * N * sizeof(float));

    cudaMemcpy(d_A, h_A.data(), M * K * sizeof(float), cudaMemcpyHostToDevice);
    cudaMemcpy(d_B, h_B.data(), K * N * sizeof(float), cudaMemcpyHostToDevice);
    cudaMemcpy(d_C, h_C.data(), M * N * sizeof(float), cudaMemcpyHostToDevice);

    dim3 blockDim(16, 16);
    dim3 gridDim((N + blockDim.x - 1) / blockDim.x,
                 (M + blockDim.y - 1) / blockDim.y);

    gemm_coalesced<float><<<gridDim, blockDim>>>(M, N, K,
                                                 1.0f,
                                                 d_A,
                                                 d_B,
                                                 0.0f,
                                                 d_C);

    cudaMemcpy(h_C.data(), d_C, M * N * sizeof(float), cudaMemcpyDeviceToHost);

    std::cout << "C =" << std::endl;
    for (int i = 0; i < M; ++i) {
        for (int j = 0; j < N; ++j) {
            std::cout << h_C[i * N + j] << " ";
        }
        std::cout << std::endl;
    }

    cudaFree(d_A);
    cudaFree(d_B);
    cudaFree(d_C);

    return 0;
}
```

## 7. 课件总结

- GEMM 可以先按 **A 矩阵的行**，也可以按 **B 矩阵的列** 进行线程映射变化。
- Naive 版本更像是按照 **A 的行** 方向变化。
- Coalesced 版本更像是按照 **B 的列** 方向变化。
